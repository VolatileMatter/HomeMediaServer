tickets.json — the backend

Create D:\Server\Dashboard\tickets.json with an empty array:

[]

In the public status page (monitor-public.html), a feedback form submits to a tiny Python endpoint that appends to this file:

# In your stats_server.py (from the Dashboard tab), add:
import json, uuid
from datetime import datetime

@app.route('/api/feedback', methods=['POST'])
def feedback():
    data = request.json
    tickets = []
    try:
        with open('D:/Server/Dashboard/tickets.json') as f:
            tickets = json.load(f)
    except: pass

    ticket = {
        'id':      'TKT-' + str(len(tickets)+1).zfill(4),
        'user':    data.get('user', 'anonymous'),
        'subject': data.get('subject', '(no subject)')[:100],
        'body':    data.get('body', '')[:2000],
        'status':  'unread',
        'date':    datetime.now().strftime('%Y-%m-%d %H:%M'),
        'ip':      request.remote_addr
    }
    tickets.append(ticket)
    with open('D:/Server/Dashboard/tickets.json','w') as f:
        json.dump(tickets, f, indent=2)
    return jsonify({'ok': True, 'id': ticket['id']})

  <input name="subject" placeholder="Subject" required>
  <textarea name="body" placeholder="Your message…" rows="4"></textarea>
  <button type="submit">Send Feedback</button>
</form>

<script>
async function submitFeedback(e) {
  e.preventDefault();
  const fd = new FormData(e.target);
  await fetch('http://[server-ip]:9001/api/feedback', {
    method: 'POST',
    headers: {'Content-Type':'application/json'},
    body: JSON.stringify(Object.fromEntries(fd))
  });
  alert('Feedback sent! Thanks.');
  e.target.reset();
}
</script>
