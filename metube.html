🐍 Part C — Create filler_downloader.py

Save at D:\Scripts\filler_downloader.py. 

#!/usr/bin/env python3
"""
filler_downloader.py — Automated Tunarr filler acquisition
Runs nightly at 1am via Windows Task Scheduler.

Pipeline per playlist:
  1. yt-dlp downloads new items (skips already-downloaded via archive file)
  2. Format filter: 480p-720p only, no vertical video, no audio-only, smallest file
  3. ffmpeg-normalize: EBU R128 loudness normalization on each new file
  4. If type=clip-collection: PySceneDetect splits video into individual clips
  5. POST Jellyfin /Library/Refresh after all playlists complete
"""

import json, os, subprocess, sys, logging, requests
from pathlib import Path
from scenedetect import open_video, SceneManager
from scenedetect.detectors import ContentDetector

# ── Paths ──────────────────────────────────────────────────────
SCRIPT_DIR  = Path(__file__).parent
CONFIG_FILE = SCRIPT_DIR / "filler_config.json"
CACHE_DIR   = SCRIPT_DIR / "metube_cache"
LOG_FILE    = SCRIPT_DIR / "filler_downloader.log"

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s  %(levelname)-8s  %(message)s",
    handlers=[
        logging.FileHandler(LOG_FILE, encoding="utf-8"),
        logging.StreamHandler(sys.stdout),
    ],
)
log = logging.getLogger(__name__)

# ── Format selector ─────────────────────────────────────────────
# Prefer 720p VP9/H264 (smallest file). Minimum 480p.
# width>=height excludes vertical/portrait video.
# vcodec!=none ensures a video stream exists (no audio-only).
# +size suffix sorts ascending by filesize within quality tier.
YTDLP_FORMAT = (
    "bestvideo[height<=720][height>=480][vcodec!=none][width>=height]"
    "+bestaudio[acodec!=none]"
    "/bestvideo[height<=720][height>=480][vcodec!=none][width>=height]"
    "/best[height<=720][height>=480][vcodec!=none][width>=height]"
)

YTDLP_EXTRA = [
    "--format-sort", "res:720,codec:vp9:h264,+size",
    "--merge-output-format", "mp4",
    "--ignore-errors",    # skip unavailable/geo-blocked items, continue playlist
    "--no-warnings",
    "--quiet",
    "--progress",
    "--add-metadata",
]


def load_config():
    with open(CONFIG_FILE, encoding="utf-8") as f:
        return json.load(f)


def archive_file(name: str) -> Path:
    CACHE_DIR.mkdir(exist_ok=True)
    return CACHE_DIR / f"{name}.txt"


def download_playlist(playlist: dict) -> list:
    name    = playlist["name"]
    folder  = Path(playlist["folder"])
    url     = playlist["url"]
    archive = archive_file(name)
    folder.mkdir(parents=True, exist_ok=True)

    output_tmpl = str(folder / "%(title)s [%(id)s].%(ext)s")
    before = set(folder.glob("*.mp4")) | set(folder.glob("*.mkv")) | set(folder.glob("*.webm"))

    cmd = [
        "yt-dlp",
        "--format", YTDLP_FORMAT,
        "--download-archive", str(archive),
        "--output", output_tmpl,
        *YTDLP_EXTRA,
        url,
    ]
    log.info(f"[{name}] Downloading from: {url}")
    result = subprocess.run(cmd)
    if result.returncode not in (0, 1):
        log.warning(f"[{name}] yt-dlp exited with code {result.returncode}")

    after = set(folder.glob("*.mp4")) | set(folder.glob("*.mkv")) | set(folder.glob("*.webm"))
    new_files = sorted(after - before)
    log.info(f"[{name}] {len(new_files)} new file(s) downloaded")
    return new_files


def normalize_audio(filepath: Path) -> bool:
    """EBU R128 loudness normalization via ffmpeg-normalize (two-pass, in-place)."""
    log.info(f"  Normalizing: {filepath.name}")
    tmp = filepath.with_suffix(".norm.mp4")
    cmd = [
        "ffmpeg-normalize", str(filepath),
        "--output", str(tmp),
        "--audio-codec", "aac",
        "--audio-bitrate", "192k",
        "--normalization-type", "ebu",
        "--target-level", "-23",
        "--loudness-range-target", "7",
        "--keep-loudness-range-target",
        "--force", "--quiet",
    ]
    r = subprocess.run(cmd)
    if r.returncode == 0 and tmp.exists():
        filepath.unlink()
        tmp.rename(filepath)
        return True
    else:
        log.warning(f"  Normalization failed for {filepath.name}; keeping original")
        if tmp.exists():
            tmp.unlink()
        return False


def run_scene_detect(filepath: Path, out_folder: Path,
                     threshold: float, min_sec: float, max_sec: float) -> list:
    """Split a clip-collection video into individual scenes using PySceneDetect."""
    log.info(f"  Scene detect: {filepath.name}")
    video = open_video(str(filepath))
    mgr   = SceneManager()
    mgr.add_detector(ContentDetector(threshold=threshold))
    mgr.detect_scenes(video)
    scenes = mgr.get_scene_list()
    log.info(f"    {len(scenes)} scenes found")

    stem  = filepath.stem
    clips = []
    for i, (start, end) in enumerate(scenes, 1):
        dur = (end - start).get_seconds()
        if dur < min_sec:
            continue
        if dur > max_sec:
            continue
        out_path = out_folder / f"{stem}__clip_{i:04d}.mp4"
        cmd = [
            "ffmpeg", "-y",
            "-i", str(filepath),
            "-ss", str(start.get_timecode()),
            "-to", str(end.get_timecode()),
            "-c:v", "libx264", "-crf", "23", "-preset", "fast",
            "-c:a", "aac", "-b:a", "192k",
            str(out_path),
        ]
        r = subprocess.run(cmd, capture_output=True)
        if r.returncode == 0:
            clips.append(out_path)
        else:
            log.warning(f"    Failed to extract scene {i}")

    log.info(f"    {len(clips)} usable clips extracted")
    return clips


def trigger_jellyfin_scan(base_url: str, api_key: str, library_id: str = "") -> None:
    if not api_key:
        log.warning("No Jellyfin API key configured — skipping scan")
        return
    endpoint = f"{base_url}/Items/{library_id}/Refresh" if library_id else f"{base_url}/Library/Refresh"
    try:
        r = requests.post(endpoint, headers={"X-Emby-Token": api_key}, timeout=15)
        if r.status_code in (200, 204):
            log.info(f"Jellyfin scan triggered (HTTP {r.status_code})")
        else:
            log.warning(f"Jellyfin scan returned HTTP {r.status_code}")
    except requests.RequestException as e:
        log.error(f"Could not reach Jellyfin: {e}")


def process_playlist(playlist: dict) -> None:
    name    = playlist["name"]
    pl_type = playlist.get("type", "standard")
    log.info(f"=== [{name}] type={pl_type} ===")

    new_files = download_playlist(playlist)
    if not new_files:
        log.info(f"[{name}] Nothing new.")
        return

    if pl_type == "clip-collection":
        out_folder   = Path(playlist["folder"])
        threshold    = playlist.get("scene_threshold",   27.0)
        min_sec      = float(playlist.get("scene_min_seconds",  3))
        max_sec      = float(playlist.get("scene_max_seconds", 300))
        for f in new_files:
            clips = run_scene_detect(f, out_folder, threshold, min_sec, max_sec)
            for clip in clips:
                normalize_audio(clip)
            log.info(f"  {f.name}: {len(clips)} clips produced")
    else:
        for f in new_files:
            normalize_audio(f)


def main():
    log.info("=" * 60)
    log.info("filler_downloader.py — run start")
    cfg = load_config()
    for playlist in cfg.get("playlists", []):
        try:
            process_playlist(playlist)
        except Exception as e:
            log.error(f"Error in [{playlist.get('name','?')}]: {e}", exc_info=True)
    log.info("Triggering Jellyfin library scan...")
    trigger_jellyfin_scan(
        cfg.get("jellyfin_base_url", "http://localhost:8096"),
        cfg.get("jellyfin_api_key", ""),
        cfg.get("jellyfin_library_id", ""),
    )
    log.info("filler_downloader.py — run complete")
    log.info("=" * 60)


if __name__ == "__main__":
    main()

Tunarr uses filler to bridge the gap between the end of content and the next fixed schedule slot. The ideal filler clip duration depends on how large the typical gap is for each slot type. Curate playlists with these ranges in mind.

Drama episode 60 min 42–55 min 5–18 min 5–10 min 3–15 min

Sweet spot: 3–10 minutes covers sitcom gaps perfectly and handles most drama gaps with 1–2 clips chained. Avoid sourcing clips over 15 minutes unless they're specifically for movie-slot padding. PySceneDetect output (from clip-collection playlists) typically produces 30s–5min chunks — ideal as micro-pads and sitcom fills.
📋 Playlist Type Reference
standard — Download as-is

Each video in the playlist becomes one filler clip. Best for: documentary shorts, video essays, ambient footage, making-of featurettes, stand-up clips. Use when content is already short (under 15 minutes) or when you want the full video intact. 

"type": "standard"
clip-collection — Scene-split

Each video is a compilation (e.g. Best Pokémon Moments Vol. 3). PySceneDetect finds scene boundaries and splits each video into individual clips. Each clip becomes its own independent filler item. Best for: AMV compilations, highlight reels, reaction compilations, moment collections. 

"type": "clip-collection",
"scene_threshold": 27.0,
"scene_min_seconds": 3,
"scene_max_seconds": 300

🛠️ Common Issues
❌ Nothing downloads — "ERROR: Unsupported URL" or "extractor not found"
yt-dlp extractor is out of date. Run yt-dlp -U and retry. This is the most common failure mode; YouTube breaks extractors within days of front-end updates.
❌ Videos download as audio-only despite the format selector
That specific video has no 480p+ video stream on YouTube (rare, but happens with some uploads). The script logs a warning and skips it. Confirm with: 
yt-dlp -F "https://youtube.com/watch?v=VIDEO_ID"
If no video stream ≥480p exists, the correct behavior is to skip it.
⚠️ PySceneDetect produces too many clips (flash-cuts, micro-clips under 1 second)
Increase scene_min_seconds (e.g. from 3 to 5) and/or increase scene_threshold (e.g. from 27 to 35) in filler_config.json for that playlist. Test with the manual detection command in the Maintenance tab before committing.
⚠️ Normalization step is very slow on first run
ffmpeg-normalize does a two-pass encode per file. A 10-minute clip takes ~30–60s; a large playlist on first run may take hours. Subsequent nightly runs only process newly downloaded files, so this is a one-time cost per video. If speed matters, reduce quality slightly: change -crf 23 to -crf 28 in the run_scene_detect ffmpeg call.
❌ Jellyfin scan fails — "Could not reach Jellyfin"
Check: (1) Jellyfin is running, (2) the API key in filler_config.json is valid. Test: 
curl http://localhost:8096/System/Info -H "X-Emby-Token: YOUR_KEY"
401 = wrong key. Connection refused = Jellyfin is down.
⚠️ Task Scheduler fires but log is empty
Python not found at the path in the task action. Run where python in CMD and update the task. Also confirm "Start in" is set to D:\Scripts.
⚠️ Vertical video still occasionally appears in the filler folder
Some videos only have portrait-mode streams; yt-dlp falls through to best-available when the format filter matches nothing. Post-process check with ffprobe: 
ffprobe -v error -select_streams v:0 \
  -show_entries stream=width,height \
  -of csv=p=0 "D:\Media\Filler\channel\somefile.mp4"
# If height > width, delete the file.
