Step 2: Create the recommendation server

This Flask endpoint loads your catalogs and user history, then calls Ollama with a structured prompt.

# D:\Server\Recommendations\rec_server.py
from flask import Flask, request, jsonify
import json, os, requests

app = Flask(__name__)

BOOKS_FILE = 'D:/Server/Recommendations/books.json'
GAMES_FILE = 'D:/Server/Recommendations/games.json'
HIST_FILE  = 'D:/Server/Recommendations/history.json'
OLLAMA_URL = 'http://localhost:11434/api/chat'
MODEL      = 'llama3.1'  # or whichever you pulled

def load_json(path):
    if os.path.exists(path):
        with open(path) as f: return json.load(f)
    return []

def get_user_history(username):
    hist = load_json(HIST_FILE)
    return hist.get(username, {'books_read': [], 'games_played': []})

def record_activity(username, item_type, item_id):
    hist = load_json(HIST_FILE)
    if username not in hist:
        hist[username] = {'books_read': [], 'games_played': []}
    key = 'books_read' if item_type == 'book' else 'games_played'
    if item_id not in hist[username][key]:
        hist[username][key].append(item_id)
    with open(HIST_FILE, 'w') as f:
        json.dump(hist, f, indent=2)

@app.route('/api/recommend', methods=['POST'])
def recommend():
    data = request.json
    username  = data.get('user', 'anonymous')
    mode      = data.get('mode', 'both')   # book / game / both
    history_msgs = data.get('messages', [])
    new_msg   = data.get('message', '')

    books  = load_json(BOOKS_FILE) if mode in ('book','both')  else []
    games  = load_json(GAMES_FILE) if mode in ('game','both')  else []
    hist   = get_user_history(username)

    system_prompt = f"""You are a personal librarian and game curator for a home media server.
You have access to the following library:

BOOKS AVAILABLE ({len(books)} total):
{json.dumps(books, indent=2)[:8000]}

RETRO GAMES AVAILABLE ({len(games)} total):
{json.dumps(games, indent=2)[:8000]}

USER HISTORY for {username}:
Books read: {hist['books_read']}
Games played: {hist['games_played']}

INSTRUCTIONS:
- Recommend items ONLY from the lists above — never invent titles
- Explain WHY you're recommending each item based on their history
- When recommending, format each item as: **Title** (Platform/Author) - reason
- Ask follow-up questions to refine if needed
- Be conversational and enthusiastic, like a knowledgeable friend
- Never recommend items already in their history unless they ask
- Respect their mood: if they want "something short", respect that constraint"""

    messages = [{"role": "system", "content": system_prompt}]
    for m in history_msgs:
        messages.append({"role": m['role'], "content": m['content']})
    messages.append({"role": "user", "content": new_msg})

    try:
        r = requests.post(OLLAMA_URL, json={
            "model": MODEL,
            "messages": messages,
            "stream": False
        }, timeout=120)
        reply = r.json()['message']['content']
        return jsonify({'ok': True, 'reply': reply})
    except Exception as e:
        return jsonify({'ok': False, 'error': str(e)}), 500

@app.route('/api/history/<username>')
def get_history(username):
    return jsonify(get_user_history(username))

@app.route('/api/history/record', methods=['POST'])
def record():
    d = request.json
    record_activity(d['user'], d['type'], d['id'])
    return jsonify({'ok': True})

if __name__ == '__main__':
    app.run(port=5051, host='0.0.0.0')

Step 4: Integrate with Calibre-Web and EmulatorJS for automatic history tracking

When a user finishes a book in Calibre-Web or completes a game in the emulation server, automatically record it to their history:

# In Calibre-Web: add a "Book Finished" notification hook:
# Admin → Notifications → add webhook → POST to http://localhost:5051/api/history/record
# Body: {"user": "{{user}}", "type": "book", "id": "{{book_id}}"}

# In emulation server.py: after a save event for a completed game:
requests.post('http://localhost:5051/api/history/record', json={
    'user': session['user'],
    'type': 'game',
    'id': game_id
